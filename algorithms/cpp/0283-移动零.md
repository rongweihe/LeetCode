思路：双指针。

代码

```c++

class Solution {
public:
    void moveZeroes(vector<int>& nums) {
        int size = nums.size();
        if(size == 0) {
          return;
        }
        //第一次遍历的时候，j指针记录非0的个数，只要是非0的统统都赋给nums[j]
        int j = 0;
        for(int i=0; i<size; ++i) {
            if(nums[i]) {
                nums[j++] = nums[i];
            }
        }
        //非0元素统计完了，剩下的都是0了
        //所以第二次遍历把末尾的元素都赋为0即可
        for(int i=j; i<size; ++i) {
            nums[i] = 0;
        }
    }
};
```
